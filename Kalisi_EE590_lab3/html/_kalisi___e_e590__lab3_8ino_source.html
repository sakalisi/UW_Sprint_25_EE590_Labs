514 // Task 6
515 // Observed Behavior: Simulates environmental sensing via an LDR, processes the readings, updates LED state, and stores averaged results.
516 // Commented Purpose: Demonstrates integration of string manipulation, dynamic memory, and circular buffers in a real-time sensing and control task.
517 // Edge Case Handling: Handles buffer wrap-around, buffer full condition, and dynamic array resizing automatically.
518 // Error Handling: Assumes sensor and LED GPIO are properly initialized elsewhere as direct register access is used.
519 *(volatile uint32_t *) TIMG_T0UPDATE_REG(0) = 1; // Latch timer value
520 uint32_t curr_time = *(volatile uint32_t *) TIMG_T0LO_REG(0); //read timer val
521
522 simulateSensorData(&cb, &processedData);
523
524 // c. Check if 1 Second has passed since last LED update.
525 // - If yes:
526 // Use most recent average as a brightness level for the LED.
527 // Apply this brightness level to the LED using PWM register logic.
528 // Reset the 1000 ms timer.
529 if(curr_time - LED_timer > COUNT) {
530 ledcWrite(LED, (&processedData)->data[(&processedData)->size - 1]);
531 LED_timer = curr_time;
532 }
533
534 // d. Check if 10 seconds have passed since the last background tasks:
535 // - If yes:
536 // Do all tasks from 2 to 5 again.
537 // Reset the 10000 ms timer.
538 if(curr_time - BACKGROUND_timer > 10 * COUNT) {
539 BACKGROUND_timer = curr_time;
540 task2();
541 task3();
542 task4();
543 task5();
544 }
545
546 // 3. Do not use any blocking function calls (no delays).
547 // 4. Do not use pinMode() or digitalWrite(); use direct register access instead.
548 // 5. Ensure that your code does not interfere with other running tasks in the .ino file.
549}
